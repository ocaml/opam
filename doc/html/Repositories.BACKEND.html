Module type Repositories.BACKEND

module type BACKEND = sig .. end
Backend signature
val init : Repositories.address -> unit
Initialize an OPAM repository in the current directory. The argument is the remote repository address.
val update : Repositories.address -> Types.Filename.Set.t
Update the OPAM repository in the current directory. Return the list of locally updated files.
val download_archive : Repositories.address -> Types.nv -> Types.filename Types.download
Download a (remote) archive file, stored on the (remote) OPAM repository, in the current repository. Return the local path to the downloaded archive.
val download_file : Types.nv -> Types.filename -> Types.filename Types.download
Download a (remote) file and return the local path to the downloaded file. As the opposite to download_archive, the downloaded file needs not to be stored on the remote repository. If needed, the function can use $repo/tmp/$nv/ to store transient states between downloads.
val download_dir : Types.nv -> Types.dirname -> Types.dirname Types.download
Download a (remote) directory and return the local path to the downloaded directory. If needed, the function can use $repo/tmp/$nv/ to store transient states between downloads.
val upload_dir : address:Repositories.address -> Types.dirname -> Types.Filename.Set.t
Upload the content of the current directory to the directory given as argument. Return the local paths corresponding to the uploaded local files.
