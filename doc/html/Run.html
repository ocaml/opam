Module Run


module Run: sig .. end
Low-level untyped system operations.
val mk_temp_dir : string -> string
mk_temp_dir str creates a tempory directory
val copy : string -> string -> unit
copy src dst copies src to dst
val link : string -> string -> unit
link src dst links src to dst
val real_path : string -> string
real_path p returns the real path associated to p: .. are expanded and relative paths become absolute.
val read : string -> string
read filename returns the contents of filename
val write : string -> string -> unit
write filename contents write contents to filename
val remove : string -> unit
remove filename removes filename. Works whether filename is a file or a directory
val remove_file : string -> unit
remove_file filename removes filename. Works only for normal files
val remove_dir : string -> unit
remove_dir filename removes filename. Works only for directory.
val chdir : string -> unit
Change the current working directory
val in_dir : string -> (unit -> 'a) -> 'a
in_dir dir fn evaluates fn in the directory dir
val files : string -> string list
files dir returns the files in the directory dir
val rec_files : string -> string list
rec_files dir returns the list of all files in dir, recursively
val ocaml_version : unit -> string option
Return the version of the current OCaml compiler. If no OCaml compiler is present in the path, then it returns None.
val directories : string -> string list
files dir returns the directories in the directory dir
type command = string list
a command is a list of words
val command : ?add_to_env:(string * string) list ->
       ?add_to_path:string list -> command -> int
command cmd executes the command cmd. Return the exit code.
val commands : ?add_to_env:(string * string) list ->
       ?add_to_path:string list -> command list -> int
commands ~add_to_path cmds executes the commands cmds in a context where $PATH contains add_to_path at the beginning. It stops whenever one command fails.
val read_command_output : ?add_to_env:(string * string) list ->
       ?add_to_path:string list -> command -> string list
read_command_output cmd executes the command cmd and return the lines from stdout
val extract : string -> string -> unit
extract filename dirname untar the archive filename to dirname
val cwd : unit -> string
Return the current working directory
val mkdir : string -> unit
Create a directory. Do not fail if the directory already exist.

File locking function


val flock : unit -> unit
flock () takes the global file lock. If the lock is already taken, sleep for 1s and then retry. Abort after 5 tentatives.
val funlock : unit -> unit
funlock () unlocks the global file lock. Work only if the current processus is the same as the one who took the lock at the first place.
val with_flock : ('a -> 'b) -> 'a -> 'b
Functional version of flock / funlock

Function used only by the switch commnand


val download : string -> string -> unit
download compiler sources. Currently support only tar.gz/tar.bz2, but should be able to support git and svn as well.
val patch : string -> unit
Apply a patch file in the current directory
